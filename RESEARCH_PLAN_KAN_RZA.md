|----------|-----------------|------------|
| –£–º–Ω–æ–∂–µ–Ω–∏–µ | –ú–æ—â–Ω–æ—Å—Ç—å S = U √ó I* | –û—Ç–¥–µ–ª—å–Ω–∞—è –≤–µ—Ç–∫–∞ |

| –ö–æ–Ω—ä—é–≥–∏—Ä–æ–≤–∞–Ω–∏–µ | –î–ª—è —Ä–∞—Å—á—ë—Ç–∞ –º–æ—â–Ω–æ—Å—Ç–∏ | torch.conj() |

    """
    KAN —Å —Ñ–∏–∑–∏—á–µ—Å–∫–∏ –∑–Ω–∞—á–∏–º—ã–º–∏ –æ–ø–µ—Ä–∞—Ü–∏—è–º–∏ –¥–ª—è –†–ó–ê.
    """
    def __init__(self, num_phases=3):
        super().__init__()
        # –í–µ—Ç–∫–∞ –ø—Ä—è–º—ã—Ö –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
        self.kan_direct = KANLinear(num_phases * 2, hidden, ...)
        
        # –í–µ—Ç–∫–∞ –º–æ—â–Ω–æ—Å—Ç–∏ (U √ó I*)
        self.kan_power = KANLinear(num_phases, hidden, ...)
        
        # –í–µ—Ç–∫–∞ —Å–æ–ø—Ä–æ—Ç–∏–≤–ª–µ–Ω–∏—è (U / I)
        self.kan_impedance = KANLinear(num_phases, hidden, ...)
        
        # –û–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ
        self.kan_combine = KANLinear(hidden * 3, output, ...)
    
    def forward(self, u, i):
        # –ü—Ä—è–º—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏
        direct = torch.cat([u.real, u.imag, i.real, i.imag], dim=-1)
        x_direct = self.kan_direct(direct)
        
        # –ú–æ—â–Ω–æ—Å—Ç—å
        s = u * torch.conj(i)
        x_power = self.kan_power(torch.cat([s.real, s.imag], dim=-1))
        
        # –°–æ–ø—Ä–æ—Ç–∏–≤–ª–µ–Ω–∏–µ
        z = u / (i + 1e-8)
        x_z = self.kan_impedance(torch.cat([z.abs(), z.angle()], dim=-1))
        
        return self.kan_combine(torch.cat([x_direct, x_power, x_z], dim=-1))
